获取打卡列表

url: /mobile/records/:workday (workday是时间戳，某天的00：00，例：2019-04-30 =》1556553600000)

method: POST

body:

{
 "ticket": "4f5b7e1f7cb54425a80d14bcbab9b3e6",
 "userId": "fb9e6a05e93c4d19bb9b5087e7c49e4a",
 "orgId": "10d71e67-3cdb-4866-b232-185e414de67c",
 "domainId": "atwork"
}

results: 标明注释的字段需要用到打卡API中

{
    "status": "ok",
    "msg": "operation success",
    "results": {
        "list": [
            {
                "status": "NotSigned",
                "type": "onDuty",    // onDuty上班，offDuty下班
                "scheduleName": "bug",
                "normalCheckTime": 1556586000000, //正常上班打卡时间
                "active": true,
                "workday": "2019-04-30", //工作日
                "update": false,
                "groupId": "5cbfd79adcb490fd08c78ade"
            },
            {
                "status": "NotSigned",
                "type": "offDuty",
                "scheduleName": "bug",

                "active": false,
                "workday": "2019-04-30",
                "update": false,

        ],
        "isRest": false,

打卡

url: /mobile/autocheckin

method: POST

body:

{
  "ticket": "594e2f011f914cf895581851d4e3d2c2",
    "userId": "fb9e6a05e93c4d19bb9b5087e7c49e4a",
    "orgId": "10d71e67-3cdb-4866-b232-185e414de67c",
    "domainId": "atwork",
    "location": {
        "latitude": 23.144045,
        "longitude": 113.347259
    },
    "macAddress": "", //无需传此值
    "checkType": "onDuty",
    "groupId": "5cbfd79adcb490fd08c78ade",//为列表接口的groupId
    "sourceType": "APP_AUTO",
    "targetTime": 1556586000000, //为列表接口的 normalCheckTime（注意，以下三个时间戳，都为Number类型，字符串类型不支持）
    "workday": 1556553600000,//为列表接口的 workday(记得转化成时间戳)
    "calendarTime": 1556553600000 //与workday传相同值就可以
}
